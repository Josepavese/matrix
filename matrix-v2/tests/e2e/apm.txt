# Setup: Create mock binaries in the test environment PATH
# so Matrix thinks npm/opencode are installed and sub-processes succeed
env PATH=$WORK/bin:$PATH
exec chmod +x $WORK/bin/npm
exec chmod +x $WORK/bin/opencode

# Verify unknown package fails cleanly
! exec matrix apm install doesntexist
stderr 'Unknown package'

# Run APM Install for opencode (should trigger our mock npm)
exec matrix apm install opencode
stdout 'Orchestrating installation of .opencode. via APM...'
stdout 'Package installed successfully.'

# Verify list shows installed package (because 'opencode' is in our mock PATH)
exec matrix apm list
stdout 'opencode'

# Run APM Uninstall 
exec matrix apm uninstall opencode
stdout 'Package uninstalled successfully.'

-- bin/npm --
#!/bin/sh
echo "Mock NPM execution successful"
-- bin/opencode --
#!/bin/sh
echo "Mock OpenCode"
